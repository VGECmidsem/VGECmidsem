<script type="module">
  import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";
  import { getFirestore, collection, addDoc, onSnapshot, query, orderBy, serverTimestamp }
    from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";

  const firebaseConfig = {
    apiKey: "AIzaSyDFDSgj-AYy6pHTiz0enlcu1gs2PjBF4bw",
    authDomain: "vgec-midsem.firebaseapp.com",
    projectId: "vgec-midsem",
    storageBucket: "vgec-midsem.firebasestorage.app",
    messagingSenderId: "329251861456",
    appId: "1:329251861456:web:9cdcb4f15e8617d292ffd5",
    measurementId: "G-D43S89YM3V"
  };

  const app = initializeApp(firebaseConfig);
  const db = getFirestore(app);

  let papers = [];
  let activeFilter = 'All';

  // ── Real-time listener: updates whenever anyone adds a paper ──
  const q = query(collection(db, "papers"), orderBy("createdAt", "desc"));
  onSnapshot(q, (snapshot) => {
    papers = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
    updateFilterBar();
    updateStats();
    renderPapers();
  });

  // ── Show loading state on start ──
  document.getElementById('papersContainer').innerHTML = `
    <div class="empty-state">
      <span class="empty-icon" style="animation: pulse 1s infinite;">⏳</span>
      <p>Loading papers...</p>
    </div>`;

  window.setFilter = function(f, el) {
    activeFilter = f;
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    el.classList.add('active');
    renderPapers();
  };

  function updateFilterBar() {
    const depts = ['All', ...new Set(papers.map(p => p.dept))];
    const bar = document.getElementById('filterBar');
    bar.innerHTML = depts.map(d =>
      `<button class="filter-btn ${d === activeFilter ? 'active' : ''}" onclick="setFilter('${d}', this)">${d}</button>`
    ).join('');
  }

  function updateStats() {
    const total = papers.length;
    document.getElementById('totalCount').textContent = `${total} paper${total !== 1 ? 's' : ''}`;
  }

  function renderPapers() {
    const search = document.getElementById('searchInput').value.toLowerCase();
    const filtered = papers.filter(p => {
      const matchFilter = activeFilter === 'All' || p.dept === activeFilter;
      const matchSearch = !search ||
        (p.title || '').toLowerCase().includes(search) ||
        (p.code || '').toLowerCase().includes(search) ||
        (p.dept || '').toLowerCase().includes(search) ||
        (p.semester || '').toLowerCase().includes(search);
      return matchFilter && matchSearch;
    });

    const container = document.getElementById('papersContainer');

    if (filtered.length === 0) {
      container.innerHTML = `
        <div class="empty-state">
          <span class="empty-icon">📄</span>
          <p>No papers found.<br>Be the first to upload one!</p>
        </div>`;
      return;
    }

    const groups = {};
    filtered.forEach(p => {
      if (!groups[p.dept]) groups[p.dept] = [];
      groups[p.dept].push(p);
    });

    container.innerHTML = Object.entries(groups).map(([dept, dPapers]) => `
      <div class="dept-group">
        <div class="dept-header">
          <span class="dept-label">${dept}</span>
          <span class="dept-tag">${dPapers.length} paper${dPapers.length !== 1 ? 's' : ''}</span>
        </div>
        <div class="papers-grid">
          ${dPapers.map(p => paperCard(p)).join('')}
        </div>
      </div>
    `).join('');
  }

  function paperCard(p) {
    const safeLink = (p.link || '').replace(/'/g, "\\'");
    return `
      <div class="paper-card">
        <div class="paper-top">
          <div class="paper-title">${p.title || p.dept + ' — ' + p.semester}</div>
          <span class="paper-type type-midsem">Mid-Sem</span>
        </div>
        <div class="paper-meta">
          <span class="meta-chip">📚 ${p.code || 'N/A'}</span>
          <span class="meta-chip">🗓 ${p.semester}</span>
          ${p.date ? `<span class="meta-chip">📅 ${p.date}</span>` : ''}
        </div>
        <div class="paper-actions">
          <a href="${p.link}" target="_blank" class="btn-view">↗ Open Paper</a>
          <button class="btn-copy" onclick="copyLink('${safeLink}', this)" title="Copy link">⎘</button>
        </div>
      </div>`;
  }

  window.copyLink = function(link, btn) {
    navigator.clipboard.writeText(link).then(() => {
      const orig = btn.textContent;
      btn.textContent = '✓';
      btn.style.color = 'var(--success)';
      setTimeout(() => { btn.textContent = orig; btn.style.color = ''; }, 1500);
    });
  };

  window.submitPaper = async function() {
    const link     = document.getElementById('paperLink').value.trim();
    const dept     = document.getElementById('dept').value;
    const semester = document.getElementById('semester').value;
    const code     = document.getElementById('subjectCode').value.trim();
    const examDate = document.getElementById('examDate').value;

    if (!link || !dept || !semester) {
      alert('Please fill in all required fields (link, department, semester).');
      return;
    }

    const btn = document.querySelector('.submit-btn');
    btn.textContent = 'Saving...';
    btn.disabled = true;

    try {
      await addDoc(collection(db, "papers"), {
        title: `${dept} — ${semester}`,
        link,
        dept,
        semester,
        code: code || 'N/A',
        type: 'Mid-Sem',
        date: examDate || '',
        createdAt: serverTimestamp()
      });

      ['paperLink','subjectCode','examDate'].forEach(id => document.getElementById(id).value = '');
      ['dept','semester'].forEach(id => document.getElementById(id).value = '');

      const toast = document.getElementById('toast');
      toast.classList.add('show');
      setTimeout(() => toast.classList.remove('show'), 3000);
    } catch (err) {
      alert('Error saving paper: ' + err.message);
    }

    btn.textContent = '+ Add Paper to Hub';
    btn.disabled = false;
